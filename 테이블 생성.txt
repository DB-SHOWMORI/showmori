CREATE TABLE USER_T (
U_PRM INT,
ID varchar(10),
PW CHAR(10),
U_NAME VARCHAR(10),
U_PHONE VARCHAR(20),

PRIMARY KEY (U_PRM)
);

CREATE TABLE POST_T(
S_PRM INT NOT NULL,
U_PRM INT NOT NULL,
S_TITLE VARCHAR(255), 
S_CAST LONGTEXT,
S_SYNOP LONGTEXT,

PRIMARY KEY(S_PRM),
FOREIGN KEY (U_PRM) REFERENCES USER_T(U_PRM)
);


CREATE TABLE S_INFO_T(
S_INFO_ID INT NOT NULL,
S_PRM INT NOT NULL,
S_GOALSUM INT DEFAULT 0,
S_DEADLINE DATE,
U_PRM INT NOT NULL,

PRIMARY KEY (S_INFO_ID),
FOREIGN KEY (U_PRM) REFERENCES USER_T (U_PRM),
FOREIGN KEY (S_PRM) REFERENCES POST_T(S_PRM)
);

CREATE TABLE S_DATE_T(
S_PRM INT NOT NULL,
DAY DATE,
TOTAL_SEAT INT DEFAULT 0,
S_DATE_PRM INT NOT NULL,

PRIMARY KEY (S_DATE_PRM),
FOREIGN KEY (S_PRM) REFERENCES POST_T(S_PRM)
);


CREATE TABLE REWARD_T(
D_PRM INT NOT NULL,
D_MONEY INT NOT NULL,
D_REWARD VARCHAR(40),
PRIMARY KEY (D_PRM)
);

CREATE TABLE D_INFO_T(
D_PRM INT NOT NULL,
S_DATE_PRM INT NOT NULL,
U_PRM INT NOT NULL,
D_MONEY INT,
D_CHECK BOOLEAN,

PRIMARY KEY (D_PRM),
FOREIGN KEY (S_DATE_PRM) REFERENCES S_DATE_T (S_DATE_PRM),
FOREIGN KEY (U_PRM) REFERENCES USER_T (U_PRM)
);
